def webscrape(url):
    response = requests.get(url)
    soup = BeautifulSoup(response.content, "html.parser")

    # Extract the company name from the URL
    company_name = re.search(r"state-rankings/(.*?)-stores-by-state", url).group(1).replace("-", " ").title()

    # Gets the first table because we are assuming same format
    table = soup.find_all("table")[0]
    function = []

    # loops through each row getting state name and number of stores
    for row in table.find_all("tr")[1:]:
        state = row.find("th").text.strip()
        stores = row.find("td").text.strip()

        function.append({
            "State": state,
            "Stores": stores,
            "Company": company_name 
        })

    # Convert list of dictionaries to a DataFrame
    function = pd.DataFrame(function)
    return function
